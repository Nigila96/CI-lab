(defun print-fibonacci ()
  (format t "Enter the number of terms: ~%")
  (let ((n (read)))
    (if (and (integerp n) (> n 0))
        (let ((a 0)
              (b 1))
          (format t "Fibonacci Series: ~%")
          (dotimes (i n)
            (format t "~d " a)
            (let ((next (+ a b)))
              (setq a b)
              (setq b next))))
        (format t "Please enter a positive integer."))))


(print-fibonacci)

(defun string-palindrome (str)
  (string= str (coerce (reverse (coerce str 'list)) 'string))
)

(format t "Enter string: ~%")
(let ((s (read-line)))
  (if (string-palindrome s)
      (format t "Palindrome~%")
      (format t "Not palindrome~%")
  )
)
